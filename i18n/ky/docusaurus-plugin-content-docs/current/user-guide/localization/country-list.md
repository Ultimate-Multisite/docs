# Ultimate Multisite мамлекеттер тизмесин кайра тартипт келтирүү

Мамлекет тандоочу (dropdown) — бул жерде мамлекет атынын орусча варианттары боюнча алфавиттик тартип менен сараланат. Бул сиздин компанияңыздын жана/же көпчүлүк кардарыңыздын мекени мамлекесинин аталышы өтө узун мамлекеттер тизмесинин ортосуна жабылып калышы мүмкүн дегенди билдиши мүмкүн.

Бул көйгөөнү чечүүгө жана тандаган мамлекеңиздин аты менен тандоо тизмесинин үстүнө коюуга жардам берүү үчүн, төмөнкү фрагментти колдоно аласыз (бул аны mu-pluginге кошуп койсоңуз болот):

add_filter('wu_get_countries', function($countries) {

// DE'ни сиздин мамлекеңиздин эки тамга кодду менен алмаштырыңыз

// Мисал Германия үчүн төмөндө.

unset($countries['DE']);

// DE'ни сиздин мамлекеңиздин эки тамга коду жана аты менен алмаштырыңыз.

// Мисал Германия үчүн төмөндө.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
