# Ultimate Multisite улсын дуусугт дахин эрэмбэлэх

Улс сонгох доод хавсралт нь улсын нэрний англи хэл дээрх хувилбаруудаар үсэг бүртэй дарааллаар эрэмбэлэгддэг. Энэ нь заримдаа танай компанийн эх орон нутаг, мөн ихэнх хэрэглэгчдийн улсыг маш урт улсын нэрийн жагсаалтын дунд хайж олох боломжтой гэсэн үг юм.

Үүнийг засах, сонгосон улсыг сонголтын жагсаалтын хамгийн дээд талд оруулж өгөхөд туслахын тулд доорх snippet-ийг ашиглаж болно (тань mu-plugin руу нэмж болно):

add_filter('wu_get_countries', function($countries) {

// DE-г танай улсын хоёр давхар код руу өөрчилнө

// Жишээ нь Германы хувьд.

unset($countries['DE']);

// DE-г танай улсын хоёр давхар код руу, мөн танай улсын нэрний хэлээр өөрчилнө.

// Жишээ нь Германы хувьд.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
