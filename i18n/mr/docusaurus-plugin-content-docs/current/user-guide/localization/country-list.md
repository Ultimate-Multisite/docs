# Ultimate Multisite देश यादीचा क्रम बदलणे

देश निवडण्यासाठीची ड्रॉपडाउन यादी इंग्रजी देशांच्या नावांवर आधारित अक्षरानुक्रमे (alphabetically) मांडलेली असते. त्यामुळे कधीकधी तुमच्या कंपनीचा किंवा बहुतांश ग्राहकांचा देश मोठ्या यादीत मध्येच कुठेतरी दडलेला असतो.

हे सोडवण्यासाठी आणि तुम्हाला हवा असलेला देश यादीच्या वर आणण्यासाठी, खालील कोड स्निपेट वापरा (तुम्ही हे mu-plugin मध्ये जोडू शकता):

add_filter('wu_get_countries', function($countries) {

// DE च्या जागी तुमच्या देशाचा दोन-अक्षरी कोड टाका

// खालील उदाहरण Germany साठी आहे.

unset($countries['DE']);

// DE च्या जागी तुमच्या देशाचा दोन-अक्षरी कोड

// आणि देशाचे नाव टाका.

// खालील उदाहरण Germany साठी आहे.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
