# Ultimate Multisite ਦੇਸ਼ ਸੂਚੀ ਨੂੰ ਮੁੜ ਕ੍ਰਮਬੱਧ ਕਰਨਾ

ਦੇਸ਼ ਚੁਣਨ ਲਈ dropdown ਦੇਸ਼ਾਂ ਦੇ ਅੰਗਰੇਜ਼ੀ ਨਾਵਾਂ ਦੇ ਅਨੁਸਾਰ ਅੱਖਰਕ੍ਰਮ ਵਿੱਚ ਵਿਵਸਥਿਤ ਹੁੰਦੀ ਹੈ। ਇਸਦਾ ਮਤਲਬ ਇਹ ਹੋ ਸਕਦਾ ਹੈ ਕਿ ਕਈ ਵਾਰ ਤੁਹਾਡੀ ਕੰਪਨੀ ਦਾ ਦੇਸ਼ ਅਤੇ/ਜਾਂ ਤੁਹਾਡੇ ਜ਼ਿਆਦਾਤਰ ਗਾਹਕਾਂ ਦਾ ਦੇਸ਼ ਇੱਕ ਲੰਬੀ ਦੇਸ਼ ਸੂਚੀ ਦੇ ਵਿਚਕਾਰ ਦੱਬਿਆ ਰਹਿੰਦਾ ਹੈ।

ਇਸ ਨੂੰ ਠੀਕ ਕਰਨ ਅਤੇ ਆਪਣੀ ਪਸੰਦ ਦਾ ਦੇਸ਼ ਸੂਚੀ ਦੇ ਸਿਖਰ 'ਤੇ ਲਿਆਉਣ ਲਈ, ਤੁਸੀਂ ਹੇਠਾਂ ਦਿੱਤੇ ਕੋਡ ਸਨਿੱਪਟ ਦੀ ਵਰਤੋਂ ਕਰ ਸਕਦੇ ਹੋ (ਤੁਸੀਂ ਇਸਨੂੰ mu-plugin ਵਿੱਚ ਜੋੜ ਸਕਦੇ ਹੋ):

add_filter('wu_get_countries', function($countries) {

// DE ਨੂੰ ਆਪਣੇ ਦੇਸ਼ ਦੇ ਦੋ-ਅੱਖਰੀ ਕੋਡ ਨਾਲ ਬਦਲੋ

// ਹੇਠਾਂ Germany ਲਈ ਉਦਾਹਰਨ ਦਿੱਤੀ ਗਈ ਹੈ।

unset($countries['DE']);

// DE ਨੂੰ ਆਪਣੇ ਦੇਸ਼ ਦੇ ਦੋ-ਅੱਖਰੀ ਕੋਡ ਨਾਲ

// ਅਤੇ ਆਪਣੇ ਦੇਸ਼ ਦੇ ਨਾਮ ਨਾਲ ਬਦਲੋ।

// ਹੇਠਾਂ Germany ਲਈ ਉਦਾਹਰਨ ਦਿੱਤੀ ਗਈ ਹੈ।

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
