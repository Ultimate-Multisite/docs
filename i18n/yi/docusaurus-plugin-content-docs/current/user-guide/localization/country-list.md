# איבעראָרדענען די Ultimate Multisite לאַנד־רשימה {#re-ordering-the-ultimate-multisite-country-list}

דער דראָפּדאַון צו אויסקלײַבן אַ לאַנד איז אָרדערט אַלפֿאַבעטיש לויט די ענגלישע וואַריאַנטן פֿון די לענדער־נעמען. דאָס קען מיינען אַז, צומאָל, דאָס היימלאַנד פֿון אײַער פֿירמע און/אָדער פֿון רובֿ פֿון אײַערע קונים איז באַהאַלטן אין מיטן פֿון אַ ריזיקער רשימה פֿון לענדער־נעמען.

כּדי אײַך צו העלפֿן דאָס פֿאַרריכטן און צולייגן אײַער אויסגעקליבן לאַנד צום אָנהייב פֿון דער אויסקלײַב־רשימה, קענט איר ניצן דעם סניפּעט אונטן (איר קענט אים צולייגן צו אַ MU plugin):

add_filter('wu_get_countries', function($countries) {

// טוישט DE צום צוויי־אותיותדיקן קאָד פֿון אײַער לאַנד

// בײַשפּיל אונטן פֿאַר דײַטשלאַנד.

unset($countries['DE']);

// טוישט DE צום צוויי־אותיותדיקן קאָד פֿון אײַער לאַנד

// און דעם נאָמען פֿון אײַער לאַנד.

// בײַשפּיל אונטן, פֿאַר דײַטשלאַנד.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
