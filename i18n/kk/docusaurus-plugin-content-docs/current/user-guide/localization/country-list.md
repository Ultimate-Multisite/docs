# Ultimate Multisite елделер тізімін қайта реттеу {#re-ordering-the-ultimate-multisite-country-list}

Елдік таңдауға арналған дропдаун (dropdown) есімдердің ағылшын нұсқалары бойынша әліппендикуляр ретімен орналастырылады. Бұл сіздің компанияңыздың немесе көптеген клиенттерінің үй елдігінің атауы ұзын ел тізімінің ортасында жабылып қалуы мүмкін дегенді білдіруі мүмкін.

Бұл мәселені шешуге және таңдау тізімінің жоғарғы орнына өз еліңді қосу үшін төмендегі фрагментті қолдануға болады (оны mu-plugin-ге қоса аласыз):

add_filter('wu_get_countries', function($countries) {

// DE-ні еліңнің екі әріптік кодына өзгертіңіз.

// Мысал Германия үшін төмендегідей.

unset($countries['DE']);

// DE-ні еліңнің екі әріптік кодына және еліңнің атауына өзгертіңіз.

// Мысал Германия үшін төмендегідей.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
