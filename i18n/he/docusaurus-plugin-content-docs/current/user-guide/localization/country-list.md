# שינוי סדר רשימת המדינות ב-Ultimate Multisite

התפריט הנפתח לבחירת מדינה מסודר לפי סדר אלפביתי על בסיס שמות המדינות באנגלית. המשמעות היא שלפעמים המדינה שבה נמצאת החברה שלכם ו/או רוב הלקוחות שלכם עלולה להיות קבורה באמצע רשימה ארוכה של שמות מדינות.

כדי לעזור לכם לתקן את זה ולהוסיף את המדינה הרצויה לראש רשימת הבחירה, תוכלו להשתמש בקטע הקוד הבא (אפשר להוסיף אותו ל-mu-plugin):

add_filter('wu_get_countries', function($countries) {

// שנו את DE לקוד בן שתי האותיות של המדינה שלכם

// הדוגמה למטה היא עבור גרמניה.

unset($countries['DE']);

// שנו את DE לקוד בן שתי האותיות של המדינה שלכם

// ואת שם המדינה שלכם.

// הדוגמה למטה היא עבור גרמניה.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
