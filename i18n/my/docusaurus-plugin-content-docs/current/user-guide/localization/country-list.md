# Ultimate Multisite နိုင်ငံစာရင်းကို အစီအစဥ်ပြန်စီခြင်း

နိုင်ငံရွေးချယ်ရန် dropdown သည် အင်္ဂလိပ်ဘာသာ နိုင်ငံအမည်များအပေါ် အခြေခံ၍ အက္ခရာစဥ်အတိုင်း စီထားပါသည်။ ဒါကြောင့် သင့်ကုမ္ပဏီရဲ့ မူရင်းနိုင်ငံ သို့မဟုတ် သင့်ဖောက်သည်အများစုရဲ့ နိုင်ငံဟာ နိုင်ငံအမည် စာရင်းရှည်ကြီးရဲ့ အလယ်မှာ ရှိနေတတ်ပါတယ်။

ဒီပြဿနာကို ဖြေရှင်းပြီး သင်လိုချင်တဲ့ နိုင်ငံကို ရွေးချယ်စရာစာရင်းရဲ့ ထိပ်ဆုံးမှာ ထည့်နိုင်ဖို့ အောက်ပါ code snippet ကို သုံးနိုင်ပါတယ် (mu-plugin မှာ ထည့်နိုင်ပါတယ်):

add_filter('wu_get_countries', function($countries) {

// DE ကို သင့်နိုင်ငံရဲ့ စာလုံးနှစ်လုံး code နဲ့ ပြောင်းပါ

// အောက်မှာ ဂျာမနီအတွက် ဥပမာပြထားပါတယ်။

unset($countries['DE']);

// DE ကို သင့်နိုင်ငံရဲ့ စာလုံးနှစ်လုံး code နဲ့ ပြောင်းပါ

// နိုင်ငံအမည်ကိုလည်း ပြောင်းပါ။

// အောက်မှာ ဂျာမနီအတွက် ဥပမာပြထားပါတယ်။

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
