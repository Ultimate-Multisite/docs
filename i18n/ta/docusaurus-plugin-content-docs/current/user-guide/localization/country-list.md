# Ultimate Multisite நாடு பட்டியலை மறுவரிசைப்படுத்துதல்

நாட்டைத் தேர்வு செய்வதற்கான dropdown, நாட்டுப் பெயர்களின் ஆங்கில வடிவங்களின் அடிப்படையில் அகர வரிசையில் வரிசைப்படுத்தப்படுகிறது. இதனால், சில நேரங்களில் உங்கள் நிறுவனத்தின் சொந்த நாடு அல்லது உங்கள் பெரும்பாலான வாடிக்கையாளர்களின் நாடு, மிகப்பெரிய நாட்டுப் பெயர் பட்டியலின் நடுவில் புதைந்து இருக்கலாம்.

இதைச் சரிசெய்து, நீங்கள் விரும்பும் நாட்டை தேர்வு பட்டியலின் மேலே சேர்க்க, கீழே உள்ள code snippet-ஐப் பயன்படுத்தலாம் (இதை ஒரு mu-plugin-இல் சேர்க்கலாம்):

add_filter('wu_get_countries', function($countries) {

// DE என்பதை உங்கள் நாட்டின் இரண்டு எழுத்து குறியீடாக மாற்றவும்

// கீழே ஜெர்மனிக்கான உதாரணம்.

unset($countries['DE']);

// DE என்பதை உங்கள் நாட்டின் இரண்டு எழுத்து குறியீடாக மாற்றவும்

// மற்றும் உங்கள் நாட்டின் பெயரையும் மாற்றவும்.

// கீழே ஜெர்மனிக்கான உதாரணம்.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
