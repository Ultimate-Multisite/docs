# ការរៀបចំលំដាប់ឡើងវិញនៃបញ្ជីប្រទេសសម្រាប់ Ultimate Multisite {#re-ordering-the-ultimate-multisite-country-list}

តារាងទម្លាក់ចុះដើម្បីជ្រើសរើសប្រទេសត្រូវបានរៀបចំតាមអក្ខរក្រមផ្អែកលើឈ្មោះប្រទេសជាភាសាអង់គ្លេស។ នេះអាចមានន័យថា ពេលខ្លះ ប្រទេសដើមរបស់ក្រុមហ៊ុនអ្នក និង/ឬអតិថិជនភាគច្រើនរបស់អ្នកអាចលាក់ខ្លួននៅកណ្តាលបញ្ជីឈ្មោះប្រទេសដ៏វែងមួយ។

ដើម្បីជួយអ្នកដោះស្រាយបញ្ហានេះ និងបន្ថែមប្រទេសដែលអ្នកចង់ជ្រើសរើសទៅខាងលើនៃបញ្ជីការជ្រើសរើស អ្នកអាចប្រើបំណែក (snippet) ខាងក្រោមនេះ (អ្នកអាចបន្ថែមវាទៅក្នុង mu-plugin បាន)៖

add_filter('wu_get_countries', function($countries) {

// ប្តូរ DE ទៅជាលេខកូដពីរអក្សរនៃប្រទេសរបស់អ្នក

// ឧទាហរណ៍ខាងក្រោមសម្រាប់ប្រទេសអាល្លឺម៉ង់។

unset($countries['DE']);

// ប្តូរ DE ទៅជាលេខកូដពីរអក្សរនៃប្រទេសរបស់អ្នក

// និងឈ្មោះប្រទេសរបស់អ្នក។

// ឧទាហរណ៍ខាងក្រោមសម្រាប់ប្រទេសអាល្លឺម៉ង់។

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
