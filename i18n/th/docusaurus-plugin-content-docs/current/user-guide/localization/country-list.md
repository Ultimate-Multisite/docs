# การจัดเรียงรายการประเทศใน Ultimate Multisite ใหม่

เมนูดรอปดาวน์สำหรับเลือกประเทศจะเรียงตามตัวอักษรภาษาอังกฤษของชื่อประเทศ ซึ่งหมายความว่าบางครั้งประเทศที่บริษัทของคุณตั้งอยู่ หรือประเทศที่ลูกค้าส่วนใหญ่ของคุณอาศัยอยู่ อาจจะถูกซ่อนอยู่กลางรายการประเทศที่ยาวเหยียด

เพื่อช่วยให้คุณแก้ไขปัญหานี้และเพิ่มประเทศที่ต้องการไว้ด้านบนสุดของรายการ คุณสามารถใช้โค้ดด้านล่างนี้ได้ (สามารถเพิ่มใน mu-plugin):

add_filter('wu_get_countries', function($countries) {

// เปลี่ยน DE เป็นรหัสสองตัวอักษรของประเทศที่คุณต้องการ

// ตัวอย่างด้านล่างสำหรับประเทศเยอรมนี

unset($countries['DE']);

// เปลี่ยน DE เป็นรหัสสองตัวอักษรของประเทศที่คุณต้องการ

// และชื่อประเทศของคุณ

// ตัวอย่างด้านล่างสำหรับประเทศเยอรมนี

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
