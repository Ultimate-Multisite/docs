# Ultimate Multisite երկրների ցուցակի վերա sắp xếpում

Երկիր ընտրելու դրոպդաբլոնը տեսանելի է անվաների անգլերական տարբերակներով ըստ ա-զ-ի կարգով։ Սա նշանակում է, որ երբեմն ձեր ընկերության կամ հաճախորդների մեծ մասի երկիրը կարող է թաքնված լինել շատ երկար երկրների անունների ցուցակի միջև։

Դա ուղղելու և ձեր ընտրած երկիրը տեսանելիության ցուցակում ամենավերևում տեղադրելու համար կարող եք օգտագործել ստորև ներկայացված կոդը (ուղղակի կարող եք այն ավելացնել mu-plugin-ի մեջ).

add_filter('wu_get_countries', function($countries) {

// DE-ն փոխեք ձեր երկրի երկու տառով կոդով

// Օրինակ՝ Գերմանիայի համար։

unset($countries['DE']);

// DE-ն փոխեք ձեր երկրի երկու տառով կոդով

// և ձեր երկրի անվանը։

// Օրինակ՝ Գերմանիայի համար։

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
