# Ultimate Multisite രാജ്യ ലിസ്റ്റ് പുനഃക്രമീകരിക്കൽ

രാജ്യം തിരഞ്ഞെടുക്കാനുള്ള ഡ്രോപ്പ്ഡൗൺ രാജ്യനാമങ്ങളുടെ ഇംഗ്ലീഷ് പതിപ്പുകൾ അടിസ്ഥാനമാക്കി അക്ഷരമാലാക്രമത്തിലാണ് ക്രമീകരിച്ചിരിക്കുന്നത്. ഇതിനർത്ഥം, ചിലപ്പോൾ നിങ്ങളുടെ കമ്പനിയുടെയും/അല്ലെങ്കിൽ നിങ്ങളുടെ ഭൂരിഭാഗം ഉപഭോക്താക്കളുടെയും സ്വന്തം രാജ്യം ഒരു വലിയ രാജ്യനാമ ലിസ്റ്റിന്റെ നടുവിൽ എവിടെയോ മറഞ്ഞിരിക്കാം എന്നാണ്.

ഇത് പരിഹരിക്കാനും നിങ്ങൾ ആഗ്രഹിക്കുന്ന രാജ്യം തിരഞ്ഞെടുപ്പ് ലിസ്റ്റിന്റെ മുകളിൽ ചേർക്കാനും, താഴെ കൊടുത്തിരിക്കുന്ന കോഡ് സ്നിപ്പറ്റ് ഉപയോഗിക്കാം (ഇത് ഒരു mu-plugin-ൽ ചേർക്കാവുന്നതാണ്):

add_filter('wu_get_countries', function($countries) {

// DE എന്നത് നിങ്ങളുടെ രാജ്യത്തിന്റെ രണ്ടക്ഷര കോഡിലേക്ക് മാറ്റുക

// താഴെ ജർമ്മനിക്കുള്ള ഉദാഹരണം.

unset($countries['DE']);

// DE എന്നത് നിങ്ങളുടെ രാജ്യത്തിന്റെ രണ്ടക്ഷര കോഡിലേക്കും

// നിങ്ങളുടെ രാജ്യത്തിന്റെ പേരിലേക്കും മാറ്റുക.

// താഴെ ജർമ്മനിക്കുള്ള ഉദാഹരണം.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
