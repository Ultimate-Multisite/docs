# Ultimate Multisite ದೇಶಗಳ ಪಟ್ಟಿಯನ್ನು ಮರುಕ್ರಮಗೊಳಿಸುವುದು

ದೇಶವನ್ನು ಆಯ್ಕೆ ಮಾಡಲು ಇರುವ dropdown ಇಂಗ್ಲಿಷ್ ಭಾಷೆಯ ದೇಶದ ಹೆಸರುಗಳ ಆಧಾರದ ಮೇಲೆ ಅಕಾರಾದಿಯಾಗಿ ವಿಂಗಡಿಸಲಾಗಿದೆ. ಇದರಿಂದಾಗಿ, ಕೆಲವೊಮ್ಮೆ ನಿಮ್ಮ ಕಂಪನಿಯ ಮತ್ತು/ಅಥವಾ ನಿಮ್ಮ ಹೆಚ್ಚಿನ ಗ್ರಾಹಕರ ಸ್ವದೇಶವು ದೊಡ್ಡ ದೇಶಗಳ ಪಟ್ಟಿಯ ಮಧ್ಯದಲ್ಲಿ ಹುದುಗಿರಬಹುದು.

ಇದನ್ನು ಸರಿಪಡಿಸಲು ಮತ್ತು ನಿಮ್ಮ ಆಯ್ಕೆಯ ದೇಶವನ್ನು ಪಟ್ಟಿಯ ಮೇಲ್ಭಾಗಕ್ಕೆ ಸೇರಿಸಲು, ಕೆಳಗಿನ ಕೋಡ್ ತುಣುಕನ್ನು ಬಳಸಬಹುದು (ಇದನ್ನು mu-plugin ಗೆ ಸೇರಿಸಬಹುದು):

add_filter('wu_get_countries', function($countries) {

// DE ಅನ್ನು ನಿಮ್ಮ ದೇಶದ ಎರಡು ಅಕ್ಷರಗಳ ಕೋಡ್‌ಗೆ ಬದಲಾಯಿಸಿ

// ಕೆಳಗಿನ ಉದಾಹರಣೆ Germany ಗಾಗಿ.

unset($countries['DE']);

// DE ಅನ್ನು ನಿಮ್ಮ ದೇಶದ ಎರಡು ಅಕ್ಷರಗಳ ಕೋಡ್‌ಗೆ ಬದಲಾಯಿಸಿ

// ಮತ್ತು ನಿಮ್ಮ ದೇಶದ ಹೆಸರನ್ನು ಬದಲಾಯಿಸಿ.

// ಕೆಳಗಿನ ಉದಾಹರಣೆ Germany ಗಾಗಿ.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
