# Зміна порядку країн у списку Ultimate Multisite

Випадний список для вибору країни відсортовано за алфавітом на основі англійських назв країн. Це означає, що іноді країна вашої компанії та/або більшості ваших клієнтів може загубитися десь посеред величезного списку.

Щоб виправити це й додати потрібну країну на початок списку вибору, скористайтеся наведеним нижче фрагментом коду (його можна додати до mu-plugin):

add_filter('wu_get_countries', function($countries) {

// Змініть DE на дволітерний код вашої країни

// Приклад нижче для Німеччини.

unset($countries['DE']);

// Змініть DE на дволітерний код вашої країни

// та назву вашої країни.

// Приклад нижче для Німеччини.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
