# Қаршии рӯйхати кишварҳои Ultimate Multisite {#re-ordering-the-ultimate-multisite-country-list}

Чӣ рӯйхати ҷудои (dropdown) барои интихоби кишвар ба таври ҳарфбуカーӣ бо варзишҳои англисӣ аз номҳои кишварҳо тартиб дода шудааст. Ин метавонад маънои онро дошта бошад, ки баъзан маҳаллаи асосии ширкати шумо ва/ё аксаки муштариёнашон дар миёна рӯйхати бузург аз номҳои кишварҳо ҷойгир аст.

Ба ёрӣ расонидани шумо барои ислоҳ кардани ин масъала ва илова кардани кишвари афзали шумо ба болои рӯйхати интихоб, метавонед аз куттиби пастфақат (snippet) зерин истифода баред (инро метавонед ба як mu-plugin илова кунед):

add_filter('wu_get_countries', function($countries) {

// DE-ро ба коди ду ҳарфӣ аз кишвари шумо тағйир диҳед

// Мисол дар паст барои Германия.

unset($countries['DE']);

// DE-ро ба коди ду ҳарфӣ аз кишвари шумо ва номи кишвари шумо тағйир диҳед.

// Мисол дар паст, барои Германия.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
