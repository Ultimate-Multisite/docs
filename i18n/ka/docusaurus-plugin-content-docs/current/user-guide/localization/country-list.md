# Ultimate Multisite ქვეყნების სია ხელახლა დალაგება {#re-ordering-the-ultimate-multisite-country-list}

ქვეყნის შერჩევის დროპდაუნი (dropdown) ინგლისური ვარიანტების მიხედვით არის დალაგებული alfabetურად. ამით შეიძლება ის იყოს, რომ ზოგჯერ თქვენი კომპანიის მთავარი ქვეყანა და/ან უმეტეს მომხმარებლების ქვეყნები ჩადებულ უზარმაზარ ქვეყნების სახელების სიაში შუაში იყოს დამალული.

იმისათვის, რომ ეს პრობლემა გადაჭიროთ და თქვენი სასურველი ქვეყანა შერჩევის სიის ზედა ნაწილში დაამატოთ, შეგიძლიათ გამოიყენოთ ქვემოთ მოცემული სნეპეტი (შეგიძლიათ დაამატოთ ის mu-plugin-ში):

add_filter('wu_get_countries', function($countries) {

// DE-ს შეცვლა თქვენი ქვეყნის ორნიშნა კოდით

// მაგალითი გერმანიისთვის.

unset($countries['DE']);

// DE-ს შეცვლა თქვენი ქვეყნის ორნიშნა კოდით

// და თქვენი ქვეყნის სახელის სახელი.

// მაგალითი გერმანიისთვის.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
