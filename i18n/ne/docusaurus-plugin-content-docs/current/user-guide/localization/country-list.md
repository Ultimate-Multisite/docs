# Ultimate Multisite देशहरूको सूचीको क्रम परिवर्तन गर्ने तरिका {#re-ordering-the-ultimate-multisite-country-list}

देश छान्ने ड्रपडाउन अंग्रेजी संस्करणका आधारमा देशका नामहरूको वर्णानुक्रममा व्यवस्थित हुन्छ। यसको मतलब यो हुन सक्छ कि कहिलेकाहीँ तपाईंको कम्पनीको गृह देश र/वा अधिकांश ग्राहकहरू कुनै ठूलो देशको नामको सूचीको बीचमा लुकेको हुन सक्छन्।

यसलाई सुधार गर्न र चयन सूचीको शीर्षमा आफ्नो रोजाइको देश थप्न मद्दत गर्न, तपाईंले तलको स्निपेट प्रयोग गर्न सक्नुहुन्छ (यो तपाईंले mu-plugin मा थप्न सक्नुहुन्छ):

add_filter('wu_get_countries', function($countries) {

// DE लाई तपाईंको देशको दुई अक्षरको कोडमा परिवर्तन गर्नुहोस्

// जर्मनीका लागि उदाहरण तल हेर्नुहोस्।

unset($countries['DE']);

// DE लाई तपाईंको देशको दुई अक्षरको कोड र नाममा परिवर्तन गर्नुहोस्।

// जर्मनीका लागि तलको उदाहरण हेर्नुहोस्।

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
