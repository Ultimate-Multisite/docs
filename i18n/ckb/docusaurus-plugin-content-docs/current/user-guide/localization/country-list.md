# دانانەکردنی لێستی وڵاتەکانی Ultimate Multisite {#re-ordering-the-ultimate-multisite-country-list}

دابەزاندنی دەفتار بۆ هەڵبژاردنی وڵات بە شێوەیەکی ئەلف반ی پێکدەهێنرێت لەسەر گۆڕانکارییە ئینگلیزییەکانی ناوەکانی وڵات. ئەمەش دەکرێت ببێت بەو کە یەکێک لە وڵاتە سەرەکیی کۆمپانیای تۆ و/یان زۆربەی کڕیارەکانت لە نێوان ناوی گەورەی ئەو لیستانەدا بگەڕێنەوە.

بۆ ئەوەی یارمەتیت بدەین ئەمە بگۆڕیت و وڵاتی خۆت بۆ سەرەوەی لیستەکە زیاد بکەیت، دەتوانیت بەپێی ئەو کۆدەی خوارەوە بەکاربهێنیت (دەتوانیت لە mu-pluginەکەت زیاد بکەیت):

add_filter('wu_get_countries', function($countries) {

// DE وەک کۆدی دوو یارانی وڵاتەکەت بگۆڕە

// نموونەی خوار بۆ ئەرمیی.

unset($countries['DE']);

// DE وەک کۆدی دوو یارانی وڵاتەکەت بگۆڕە

// و ناوی ناوچەکەت بگۆڕە.

// نموونەی خوار، بۆ ئەرمیی.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
