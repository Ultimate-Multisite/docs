# бһҳбһӮбҹ’бһӮбһ»бһ‘бҹ’бһ‘бҹҒбһҹбһҖбҹҚбһҖбһ¶бһҡбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣ

бһҳбһӮбҹ’бһӮбһ»бһ‘бҹ’бһ‘бҹҒбһҹбһҖбҹҚбһ“бҹҒбҹҮбһӮбҹ’бһҡбһ”бһҠбһҺбҹ’бһҠбһ”бҹӢбһӣбҹҶбһ“бһ¶бҹҶбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣбһ‘бһјбһ‘бҹ…бһҮбһ¶бһҳбһҪбһҷ Ultimate Multisite бһҡбһҪбһҳбһ‘бһ¶бҹҶбһ„бһҖбһ¶бһҡбһ—бҹ’бһҮбһ¶бһ”бҹӢбһ‘бҹ…бһҹбҹҒбһңбһ¶бһҖбһҳбҹ’бһҳбһҒбһ¶бһ„бһҖбҹ’бһҡбҹ… бһҖбһ¶бһҡбһ”бһ„бҹ’бһҖбһҫбһҸбһ…бҹ’бһҡбһҖбһ‘бһјбһ‘бһ¶бһҸбҹӢбһ•бҹ’бһ‘бһ¶бһӣбҹӢбһҒбҹ’бһӣбһҪбһ“ бһ“бһ·бһ„бһҖбһ¶бһҡбһҠбҹ„бҹҮбһҹбҹ’бһҡбһ¶бһҷ webhooksбҹ”

бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ бҹҒбһҠбҹ’бһӢбһ¶бһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’бһўбҹ’бһ“бһҖбһҮбһҪбһӣбһҠбһ¶бһ…бҹӢбһҠбҹ„бһҷбһЎбҹӮбһҖ бһҹбһјбһҳбһҳбһҫбһӣ [бһҖбһ¶бһҡбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣ Multi-Tenancy](./multi-tenancy) бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҖбһ¶бһҡбһҺбҹӮбһ“бһ¶бҹҶбһўбҹҶбһ–бһёбһҖбһ¶бһҡбһ…бһ¶бһ”бҹӢбһ•бҹ’бһҠбһҫбһҳбһўбҹ’бһ“бһҖбһҮбһҪбһӣбһўбһ’бһ·бһ”бһҸбҹҒбһҷбҹ’бһҷ бһҖбһ¶бһҡбһ•бҹ’бһ‘бҹҖбһ„бһ•бҹ’бһ‘бһ¶бһҸбҹӢбһҖбһ¶бһҡбһ•бҹ’бһ‘бҹҒбһҡ SSO бһ“бһ·бһ„бһҖбһ¶бһҡбһӣбһ»бһ”бһ…бҹ„бһӣбҹ”

## бһҖбһ¶бһҡбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣ CRM {#crm-integration}

бһ’бҹ’бһңбһҫбһҹбһҳбһҖбһ¶бһӣбһҖбһҳбҹ’бһҳбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһўбһҸбһ·бһҗбһ·бһҮбһ“бһ‘бҹ… CRM бһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖ бһ“бҹ…бһ–бҹҒбһӣбһўбһҸбһ·бһҗбһ·бһҮбһ“бһҗбҹ’бһҳбһёбһ…бһ»бҹҮбһҲбҹ’бһҳбҹ„бҹҮбҹ–

```php
add_action('wu_customer_post_create', 'sync_customer_to_crm');

function sync_customer_to_crm($customer) {
    $crm_api = new Your_CRM_API();

    $crm_api->create_contact([
        'email' => $customer->get_email(),
        'name' => $customer->get_display_name(),
        'signup_date' => $customer->get_date_registered(),
        'plan' => $customer->get_membership()->get_plan()->get_name()
    ]);

    // Store CRM ID for future reference
    $customer->add_meta('crm_contact_id', $crm_api->get_last_contact_id());
}
```

## бһҖбһ¶бһҡбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣбһңбһ·бһ—бһ¶бһӮбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷ {#analytics-integration}

бһҸбһ¶бһҳбһҠбһ¶бһ“бһ–бҹ’бһҡбһ№бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡбһҺбҹҚбһўбһ¶бһҮбһёбһңбһҖбһҳбҹ’бһҳбһҹбҹҶбһҒбһ¶бһ“бҹӢбҹ—бһ“бҹ…бһ‘бһјбһ‘бһ¶бҹҶбһ„бһңбһҠбҹ’бһҸбһҮбһёбһңбһ·бһҸбһўбһҸбһ·бһҗбһ·бһҮбһ“бҹ–

```php
add_action('wu_checkout_completed', 'track_conversion', 10, 3);
add_action('wu_membership_status_to_cancelled', 'track_churn');
add_action('wu_payment_failed', 'track_payment_failure');

function track_conversion($payment, $customer, $membership) {
    // Google Analytics 4
    gtag('event', 'purchase', [
        'transaction_id' => $payment->get_id(),
        'value' => $payment->get_total(),
        'currency' => $payment->get_currency(),
        'items' => [
            [
                'item_id' => $membership->get_plan()->get_id(),
                'item_name' => $membership->get_plan()->get_name(),
                'category' => 'subscription',
                'quantity' => 1,
                'price' => $payment->get_total()
            ]
        ]
    ]);
}
```
