# Ultimate Multisite દેશ યાદીને ફરીથી ગોઠવવી

દેશ પસંદ કરવા માટેનું dropdown દેશના નામોના અંગ્રેજી સંસ્કરણોના આધારે મૂળાક્ષરો પ્રમાણે ગોઠવાયેલું છે. આનો અર્થ એ થઈ શકે કે, ક્યારેક, તમારી કંપનીનો મૂળ દેશ અને/અથવા તમારા મોટાભાગના ગ્રાહકોનો દેશ દેશના નામોની મોટી યાદીની વચ્ચે ક્યાંક દટાઈ ગયેલો હોય.

આને ઠીક કરવામાં અને તમારી પસંદગીના દેશને પસંદગી યાદીની ટોચ પર ઉમેરવામાં મદદ કરવા માટે, તમે નીચેનો snippet વાપરી શકો છો (તમે તેને mu-plugin માં ઉમેરી શકો છો):

add_filter('wu_get_countries', function($countries) {

// DE ને તમારા દેશના બે-અક્ષરના કોડથી બદલો

// નીચે Germany માટેનું ઉદાહરણ છે.

unset($countries['DE']);

// DE ને તમારા દેશના બે-અક્ષરના કોડથી

// અને તમારા દેશના નામથી બદલો.

// નીચે Germany માટેનું ઉદાહરણ છે.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
