# Ultimate Multisite દેશಗಳ ಪಟ್ಟಿಯನ್ನು ಮರು-வரிசைப்படுத்துதல் {#re-ordering-the-ultimate-multisite-country-list}

ದೇಶವನ್ನು ಆಯ್ಕೆ ಮಾಡಲು ಇರುವ ಡ್ರಾಪ್‌ಡೌನ್ (dropdown) ದೇಶದ ಇಂಗ್ಲಿಷ್ ರೂಪದ ಹೆಸರಿನ ಆಧಾರದ ಮೇಲೆ ವರ್ಣಮಾಲೆಯ ಕ್ರಮದಲ್ಲಿ ಜೋಡக்கப்பட்டுள்ளது. ಇದರ ಅರ್ಥವೇನೆಂದರೆ, ಕೆಲವೊಮ್ಮೆ ನಿಮ್ಮ ಕಂಪನಿಯ ಮತ್ತು/ಅಥವಾ ಹೆಚ್ಚಿನ ಗ್ರಾಹಕರ ಪ್ರಮುಖ ದೇಶದ ಹೆಸರು ಬಹಳ ದೊಡ್ಡ ದೇಶಗಳ ಪಟ್ಟಿಯ ಮಧ್ಯೆ ಸಿಕ್ಕಿಬಿದ್ದಿರಬಹುದು.

ಇದನ್ನು ಸರಿಪಡಿಸಲು ಮತ್ತು ನೀವು ಆಯ್ಕೆ ಮಾಡಲು ಬಯಸುವ ದೇಶವನ್ನು ಆಯ್ಕೆ ಮಾಡುವ ಪಟ್ಟಿಯ ಮೊದಲ ಸ್ಥಾನದಲ್ಲಿ ಸೇರಿಸಲು, ನೀವು ಕೆಳಗಿನ ಸ್ನಿಪ್ಪೆಟ್ (snippet) ಅನ್ನು ಬಳಸಬಹುದು (ಇದನ್ನು mu-plugin ಗೆ ಸೇರಿಸಬಹುದು):

add_filter('wu_get_countries', function($countries) {

// DE ಅನ್ನು ನಿಮ್ಮ ದೇಶದ ಎರಡು ಅಕ್ಷರ ಕೋಡ್‌ಗೆ ಬದಲಾಯಿಸಿ

// ಜರ್ಮನಿಯ ಉದಾಹರಣೆಯಾಗಿ ಕೆಳಗಿನದು.

unset($countries['DE']);

// DE ಅನ್ನು ನಿಮ್ಮ ದೇಶದ ಎರಡು ಅಕ್ಷರ ಕೋಡ್ ಮತ್ತು ನಿಮ್ಮ ದೇಶದ ಹೆಸರಿಗೆ ಬದಲಾಯಿಸಿ.

// ಜರ್ಮನಿಯ ಉದಾಹರಣೆಯಾಗಿ ಕೆಳಗಿನದು.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
