# Ultimate Multisite-এ দেশের তালিকা পুনর্বিন্যাস করা

দেশ নির্বাচনের ড্রপডাউন মেনুটি ইংরেজি দেশের নামের বর্ণানুক্রম অনুসারে সাজানো থাকে। এর ফলে কখনো কখনো আপনার কোম্পানির অথবা আপনার বেশিরভাগ গ্রাহকের দেশটি বিশাল তালিকার মাঝখানে চাপা পড়ে যেতে পারে।

এই সমস্যা সমাধান করে আপনার পছন্দের দেশটি তালিকার শুরুতে আনতে নিচের কোড স্নিপেটটি ব্যবহার করতে পারেন (এটি একটি mu-plugin-এ যোগ করুন):

add_filter('wu_get_countries', function($countries) {

// DE-এর জায়গায় আপনার দেশের দুই অক্ষরের কোড দিন

// নিচে জার্মানির উদাহরণ দেওয়া হলো।

unset($countries['DE']);

// DE-এর জায়গায় আপনার দেশের দুই অক্ষরের কোড দিন

// এবং আপনার দেশের নাম লিখুন।

// নিচে জার্মানির উদাহরণ দেওয়া হলো।

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
