# دی ترتیب د لیستی ی箇ریا گۆڕین بۆ Ultimate Multisite {#re-ordering-the-ultimate-multisite-country-list}

د دراپ‌داپنی هەڵبژاردنی وڵات، بەپێی فەرموونی ئەلفبایی ناوەکانی وڵات بە زمانی ئینگلیزی ڕیزبەندی دەکرێت. ئەمەش دەتوانێت ببێت ببێت بەوەی، هەندێک جار، وڵاتی سەرەکی کۆمپانیای تۆ و/یان زۆربەی کڕیەکانیش لە نێوان لیستێکی گەورەی ناوچەکاندا بگرێتەوە.

بۆ ئەوەی یارمەتیت بدەین ئەمە بگۆڕیت و وڵاتە ئارەزوومەکانت بۆ سەرەوەی لیستەکە زیاد بکەیت، دەتوانیت بەپێی ئەو کۆدەی خوارەوە بەکاربهێنیت (دەتوانیت لە mu-plugin یەکدا زیاد بکەیت):

add_filter('wu_get_countries', function($countries) {

// DE وڵاتەکە بگۆڕە بۆ کۆدی دوو پیتریی ئەو وڵاتەی کە تۆ دەکەیت.

// نموونەی خوارەوە بۆ ئەرمیی.

unset($countries['DE']);

// DE وڵاتەکە بگۆڕە بۆ کۆدی دوو پیتریی ئەو وڵاتەی کە تۆ دەکەیت

// و ناوی ناوچەکەت.

// نموونەی خوارەوە، بۆ ئەرمیی.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
