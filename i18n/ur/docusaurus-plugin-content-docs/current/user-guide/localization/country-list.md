# Ultimate Multisite ممالک کی فہرست کی ترتیب بدلنا

ملک منتخب کرنے کا dropdown ممالک کے انگریزی ناموں کی حروف تہجی کے مطابق ترتیب دیا گیا ہے۔ اس کا مطلب یہ ہے کہ بعض اوقات آپ کی کمپنی کا ملک یا آپ کے زیادہ تر گاہکوں کا ملک ممالک کی طویل فہرست میں کہیں درمیان میں دب کر رہ جاتا ہے۔

اس مسئلے کو حل کرنے اور اپنی پسند کا ملک فہرست میں سب سے اوپر لانے کے لیے، آپ نیچے دیا گیا کوڈ استعمال کر سکتے ہیں (اسے آپ mu-plugin میں شامل کر سکتے ہیں):

add_filter('wu_get_countries', function($countries) {

// DE کو اپنے ملک کے دو حروفی کوڈ سے بدل دیں

// نیچے جرمنی کی مثال دی گئی ہے۔

unset($countries['DE']);

// DE کو اپنے ملک کے دو حروفی کوڈ سے

// اور اپنے ملک کے نام سے بدل دیں۔

// نیچے جرمنی کی مثال دی گئی ہے۔

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
