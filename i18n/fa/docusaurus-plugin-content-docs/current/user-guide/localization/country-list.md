# تغییر ترتیب لیست کشورها در Ultimate Multisite

منوی کشویی انتخاب کشور بر اساس حروف الفبای انگلیسی نام کشورها مرتب شده است. این یعنی گاهی کشور محل شرکت شما یا کشوری که بیشتر مشتریانتان در آن هستند، در میان لیست بلندبالای کشورها گم می‌شود.

برای حل این مشکل و آوردن کشور مورد نظرتان به ابتدای لیست، می‌توانید از کد زیر استفاده کنید (این کد را می‌توانید در یک mu-plugin قرار دهید):

add_filter('wu_get_countries', function($countries) {

// کد دو حرفی کشور مورد نظرتان را جایگزین DE کنید

// مثال زیر برای آلمان است.

unset($countries['DE']);

// کد دو حرفی و نام کشور مورد نظرتان را جایگزین کنید

// مثال زیر برای آلمان است.

$countries = array_merge(array( 'DE' => __('Germany', 'ultimate-multisite'), ), $countries);

return $countries;

});
